Инапсуляция-скрывает логику работы класса внутри него самого, предоставляя внешний доступ только в виде интерфейса.

function Counter() { let count = 0; this.increment = function() { count++; }; this.getCount = function() { return count; }; } let counter = new Counter(); // Увеличиваем счетчик counter.increment(); // Получаем значение счетчика console.log(counter.getCount()); // Вывод: 1

Наследование-создание нового класса на основе родительского, который наследует все свойства и методы родителя

// Базовый класс Animal function Animal(name) { this.name = name; } Animal.prototype.eat = function() { console.log(`${this.name} ест`); }; // Класс Ujil, наследующий от Animal function Ujil(name) { Animal.call(this, name); } Ujil.prototype = Object.create(Animal.prototype); Ujil.prototype.bark = function() { console.log(`${this.name} лает`); }; // Создаем новый объект Ujil let Ujil = new Ujil("Юджил"); // Едим Ujil.eat(); // Вывод: Юджил ест // Лаем Ujil.bark(); // Вывод: Юджил лает

Полиморфизм-Одинаковые интерфейсы скрывают за собой разную реализацию

function makeSound(animal) { animal.sound(); } let Ujil = { sound: function() { console.log("The Ujil barks"); } }; let Bomba = { sound: function() { console.log("The Bomba meows"); } }; makeSound(Ujil); // Output: The Ujil barks makeSound(Bomba); // Output: The Bomba meows
